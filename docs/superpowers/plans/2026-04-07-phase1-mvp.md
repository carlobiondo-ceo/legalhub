# AudienceServ Legal Hub — Phase 1 MVP Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a working legal case management platform with case CRUD, opt-in requests, document uploads, dashboard, and Google OAuth — all running in Docker Compose.

**Architecture:** Node.js Express API backend with Prisma ORM + PostgreSQL database + Next.js 14 frontend with shadcn/ui. Three Docker containers (api, web, db) orchestrated via Docker Compose. Local filesystem for file uploads via Docker volume.

**Tech Stack:** Node.js 20, Express, TypeScript, Prisma, PostgreSQL 16, Next.js 14 (App Router), Tailwind CSS, shadcn/ui, Passport.js (Google OAuth), multer (file uploads), Docker Compose

```typescript
import { Request, Response, NextFunction } from "express";

export function requireAuth(req: Request, res: Response, next: NextFunction) {
  if (!req.session?.userId) {
    return res.status(401).json({ error: "Unauthorized" });
  }
  next();
}

export function requireAdmin(req: Request, res: Response, next: NextFunction) {
  if (!req.session?.userId) {
    return res.status(401).json({ error: "Unauthorized" });
  }
  if (req.session.userRole !== "admin") {
    return res.status(403).json({ error: "Forbidden: admin access required" });
  }
  next();
}
```

- [ ] **Step 3: Create `api/src/routes/auth.ts`**

```typescript
import { Router } from "express";
import passport from "passport";
import { Strategy as GoogleStrategy } from "passport-google-oauth20";
import { PrismaClient } from "@prisma/client";
import { config } from "../config";

const router = Router();
const prisma = new PrismaClient();

passport.use(
  new GoogleStrategy(
    {
      clientID: config.google.clientId,
      clientSecret: config.google.clientSecret,
      callbackURL: config.google.callbackUrl,
    },
    async (_accessToken, _refreshToken, profile, done) => {
      try {
```

```typescript
  if (params.riskLevel?.length) {

  if (params.dateFrom || params.dateTo) {

    data: updateData,
```

```typescript
import { PrismaClient } from "@prisma/client";

const prisma = new PrismaClient();

export async function getStats() {
  const now = new Date();
  const startOfMonth = new Date(now.getFullYear(), now.getMonth(), 1);

  const [urgent, open, inProgress, resolvedThisMonth, delayed, pendingOptIns] =
    await Promise.all([
      // Urgent: escalated_urgent status OR overdue deadlines
      prisma.legalCase.count({
        where: {
          OR: [
            { status: "escalated_urgent" },
            {
              responseDeadline: { lt: now },
              status: {
                notIn: ["resolved_no_action", "resolved_settlement", "archived"],
              },
            },
          ],
        },
      }),
      // Open: all active cases (not resolved/archived)
      prisma.legalCase.count({
        where: {
          status: {
            notIn: ["resolved_no_action", "resolved_settlement", "archived"],
          },
        },
      }),
      // In Progress: specific statuses
      prisma.legalCase.count({
        where: {
          status: { in: ["in_review", "opt_in_requested", "with_lawyer"] },
        },
      }),
      // Resolved this month
      prisma.legalCase.count({
        where: {
          status: { in: ["resolved_no_action", "resolved_settlement"] },
          updatedAt: { gte: startOfMonth },
        },
      }),
      // Delayed: no activity in 7+ days, still open
      prisma.legalCase.count({
        where: {
          status: {
            notIn: ["resolved_no_action", "resolved_settlement", "archived"],
          },
          updatedAt: {
            lt: new Date(now.getTime() - 7 * 24 * 60 * 60 * 1000),
          },
        },
      }),
      // Pending opt-ins
      prisma.optInRequest.count({
        where: { status: "pending" },
      }),
    ]);

  return { urgent, open, inProgress, resolvedThisMonth, delayed, pendingOptIns };
}

export async function getUpcomingDeadlines() {
  const now = new Date();
  const in14Days = new Date(now.getTime() + 14 * 24 * 60 * 60 * 1000);

  const cases = await prisma.legalCase.findMany({
    where: {
      status: {
        notIn: ["resolved_no_action", "resolved_settlement", "archived"],
      },
      OR: [
        { responseDeadline: { lte: in14Days } },
        { internalDeadline: { lte: in14Days } },
        { followUpDate: { lte: in14Days } },
      ],
    },
    select: {
      id: true,
      caseId: true,
      title: true,
      status: true,
      responseDeadline: true,
      internalDeadline: true,
      followUpDate: true,
      assignedTo: { select: { id: true, name: true } },
    },
    orderBy: { responseDeadline: "asc" },
  });

  return cases.map((c) => {
    // Find the nearest deadline
    const deadlines = [c.responseDeadline, c.internalDeadline, c.followUpDate]
      .filter(Boolean)
      .sort((a, b) => a!.getTime() - b!.getTime());

    const nearestDeadline = deadlines[0];
    const daysUntil = nearestDeadline
      ? Math.ceil((nearestDeadline.getTime() - now.getTime()) / (1000 * 60 * 60 * 24))
      : null;

    let urgency: "overdue" | "today" | "soon" | "upcoming" = "upcoming";
    if (daysUntil !== null) {
      if (daysUntil < 0) urgency = "overdue";
      else if (daysUntil === 0) urgency = "today";
      else if (daysUntil <= 3) urgency = "soon";
    }

    return { ...c, nearestDeadline, daysUntil, urgency };
  }).sort((a, b) => {
    const order = { overdue: 0, today: 1, soon: 2, upcoming: 3 };
    return order[a.urgency] - order[b.urgency];
  });
}
```

```tsx
  field_updated: Edit,

function formatAction(log: ActivityLog): string {
  const details = log.details as Record<string, any> | null;
  switch (log.action) {
    case "case_created":
      return `created case ${details?.caseId || ""}`;
    case "status_changed":
      return `changed status from ${details?.from} to ${details?.to}`;
    case "document_uploaded":
      return `uploaded ${details?.fileName || "a document"}`;
    case "field_updated":
      return `updated ${(details?.fields as string[])?.join(", ") || "fields"}`;
    case "opt_in_created":
      return `created opt-in request ${details?.requestId || ""}`;
    case "case_deleted":
      return `deleted case ${details?.caseId || ""}`;
    default:
      return log.action.replace(/_/g, " ");
  }
}

function timeAgo(dateStr: string): string {
  const diff = Date.now() - new Date(dateStr).getTime();
  const minutes = Math.floor(diff / 60000);
  if (minutes < 1) return "just now";
  if (minutes < 60) return `${minutes}m ago`;
  const hours = Math.floor(minutes / 60);
  if (hours < 24) return `${hours}h ago`;
  const days = Math.floor(hours / 24);
  return `${days}d ago`;
}
```

```tsx
interface DeadlineStatusProps {
  deadline: string | null;
}

export default function DeadlineStatus({ deadline }: DeadlineStatusProps) {
  if (!deadline) return <span className="text-xs text-gray-400">—</span>;

  const now = new Date();
  const dl = new Date(deadline);
  const daysUntil = Math.ceil((dl.getTime() - now.getTime()) / (1000 * 60 * 60 * 24));

  if (daysUntil < 0) {
    return <span className="text-xs font-medium text-red-600">Overdue ({Math.abs(daysUntil)}d)</span>;
  }
  if (daysUntil <= 3) {
    return <span className="text-xs font-medium text-yellow-600">Due Soon ({daysUntil}d)</span>;
  }
  return <span className="text-xs text-green-600">On Track ({daysUntil}d)</span>;
```

```tsx
"use client";

import { useState } from "react";
import { Search, Filter, X } from "lucide-react";
import type { CaseStatus, CaseType, RiskLevel } from "@/lib/types";

interface CaseFiltersProps {
  onFilter: (filters: Record<string, string>) => void;
}

const statusOptions: { value: CaseStatus; label: string }[] = [
  { value: "new", label: "New" },
  { value: "in_review", label: "In Review" },
  { value: "opt_in_requested", label: "Opt-In Requested" },
  { value: "opt_in_received", label: "Opt-In Received" },
  { value: "with_lawyer", label: "With Lawyer" },
  { value: "response_sent", label: "Response Sent" },
  { value: "awaiting_reply", label: "Awaiting Reply" },
  { value: "escalated_urgent", label: "Urgent" },
  { value: "delayed", label: "Delayed" },
  { value: "resolved_no_action", label: "Resolved" },
  { value: "resolved_settlement", label: "Settlement" },
  { value: "archived", label: "Archived" },
];

const caseTypeOptions: { value: CaseType; label: string }[] = [
  { value: "spam_complaint", label: "Spam Complaint" },
  { value: "cease_and_desist", label: "Cease & Desist" },
  { value: "gdpr_request", label: "GDPR Request" },
  { value: "data_erasure_request", label: "Data Erasure" },
  { value: "litigation_threat", label: "Litigation Threat" },
  { value: "regulatory_inquiry", label: "Regulatory Inquiry" },
  { value: "other", label: "Other" },
];

const riskOptions: { value: RiskLevel; label: string }[] = [
  { value: "low", label: "Low" },
  { value: "medium", label: "Medium" },
  { value: "high", label: "High" },
  { value: "critical", label: "Critical" },
];

export default function CaseFilters({ onFilter }: CaseFiltersProps) {
  const [search, setSearch] = useState("");
  const [showFilters, setShowFilters] = useState(false);
  const [selectedStatus, setSelectedStatus] = useState<string[]>([]);
  const [selectedType, setSelectedType] = useState<string[]>([]);
  const [selectedRisk, setSelectedRisk] = useState<string[]>([]);

  const applyFilters = () => {
    const filters: Record<string, string> = {};
    if (search) filters.search = search;
    if (selectedStatus.length) filters.status = selectedStatus.join(",");
    if (selectedType.length) filters.caseType = selectedType.join(",");
    if (selectedRisk.length) filters.riskLevel = selectedRisk.join(",");
    onFilter(filters);
  };

  const clearFilters = () => {
    setSearch("");
    setSelectedStatus([]);
    setSelectedType([]);
    setSelectedRisk([]);
    onFilter({});
  };

  const toggleItem = (arr: string[], setArr: (v: string[]) => void, value: string) => {
    setArr(arr.includes(value) ? arr.filter((v) => v !== value) : [...arr, value]);
  };

  return (
    <div className="space-y-3">
      <div className="flex items-center gap-3">
        <div className="relative flex-1 max-w-md">
          <Search size={16} className="absolute left-3 top-1/2 -translate-y-1/2 text-gray-400" />
          <input
            type="text"
            placeholder="Search by email, case ID, lawyer..."
            value={search}
            onChange={(e) => setSearch(e.target.value)}
            onKeyDown={(e) => e.key === "Enter" && applyFilters()}
            className="w-full pl-9 pr-3 py-2 rounded-lg border border-gray-200 text-sm focus:outline-none focus:ring-2 focus:ring-primary/20 focus:border-primary"
          />
        </div>
        <button
          onClick={() => setShowFilters(!showFilters)}
          className="flex items-center gap-2 px-3 py-2 rounded-lg border border-gray-200 text-sm text-gray-600 hover:bg-gray-50 cursor-pointer transition-colors duration-200"
        >
          <Filter size={16} />
          Filters
        </button>
        <button
          onClick={applyFilters}
          className="px-4 py-2 rounded-lg bg-primary text-white text-sm font-medium cursor-pointer hover:bg-green-600 transition-colors duration-200"
        >
          Search
        </button>
        {(search || selectedStatus.length || selectedType.length || selectedRisk.length) > 0 && (
          <button
            onClick={clearFilters}
            className="flex items-center gap-1 px-3 py-2 rounded-lg text-sm text-gray-500 hover:text-gray-700 cursor-pointer"
          >
            <X size={14} />
            Clear
          </button>
        )}
      </div>

      {showFilters && (
        <div className="bg-white rounded-lg border border-gray-200 p-4 grid grid-cols-1 md:grid-cols-3 gap-4">
          <div>
            <label className="text-xs font-medium text-gray-500 uppercase mb-2 block">Status</label>
            <div className="flex flex-wrap gap-1.5">
              {statusOptions.map((opt) => (
                <button
                  key={opt.value}
                  onClick={() => toggleItem(selectedStatus, setSelectedStatus, opt.value)}
                  className={`px-2 py-1 rounded text-xs cursor-pointer transition-colors ${
                    selectedStatus.includes(opt.value)
                      ? "bg-primary text-white"
                      : "bg-gray-100 text-gray-600 hover:bg-gray-200"
                  }`}
                >
                  {opt.label}
                </button>
              ))}
            </div>
          </div>
          <div>
            <label className="text-xs font-medium text-gray-500 uppercase mb-2 block">Case Type</label>
            <div className="flex flex-wrap gap-1.5">
              {caseTypeOptions.map((opt) => (
                <button
                  key={opt.value}
                  onClick={() => toggleItem(selectedType, setSelectedType, opt.value)}
                  className={`px-2 py-1 rounded text-xs cursor-pointer transition-colors ${
                    selectedType.includes(opt.value)
                      ? "bg-primary text-white"
                      : "bg-gray-100 text-gray-600 hover:bg-gray-200"
                  }`}
                >
                  {opt.label}
                </button>
              ))}
            </div>
          </div>
          <div>
            <label className="text-xs font-medium text-gray-500 uppercase mb-2 block">Risk Level</label>
            <div className="flex flex-wrap gap-1.5">
              {riskOptions.map((opt) => (
                <button
                  key={opt.value}
                  onClick={() => toggleItem(selectedRisk, setSelectedRisk, opt.value)}
                  className={`px-2 py-1 rounded text-xs cursor-pointer transition-colors ${
                    selectedRisk.includes(opt.value)
                      ? "bg-primary text-white"
                      : "bg-gray-100 text-gray-600 hover:bg-gray-200"
                  }`}
                >
                  {opt.label}
                </button>
              ))}
            </div>
          </div>
        </div>
      )}
    </div>
  );
}
```

- [ ] **Step 3: Create `web/src/app/(authenticated)/cases/page.tsx`**

```tsx
"use client";

import { useEffect, useState } from "react";
import Link from "next/link";
import { Plus } from "lucide-react";
import { cases as casesApi } from "@/lib/api";
import type { LegalCase } from "@/lib/types";
import CaseTable from "@/components/cases/CaseTable";
import CaseFilters from "@/components/cases/CaseFilters";

export default function CasesPage() {
  const [caseList, setCaseList] = useState<LegalCase[]>([]);
  const [loading, setLoading] = useState(true);
  const [total, setTotal] = useState(0);

  const loadCases = (filters: Record<string, string> = {}) => {
    setLoading(true);
    casesApi
      .list(filters)
      .then((res) => {
        setCaseList(res.cases || []);
        setTotal(res.total);
      })
      .finally(() => setLoading(false));
  };

  useEffect(() => {
    loadCases();
  }, []);

  return (
    <div className="space-y-4">
      <div className="flex items-center justify-between">
        <div>
          <h2 className="text-lg font-semibold text-gray-900">Legal Cases</h2>
          <p className="text-sm text-gray-500">{total} total cases</p>
        </div>
        <Link
          href="/cases/new"
          className="flex items-center gap-2 px-4 py-2 rounded-lg bg-primary text-white text-sm font-medium cursor-pointer hover:bg-green-600 transition-colors duration-200"
        >
          <Plus size={16} />
          New Case
        </Link>
      </div>

      <CaseFilters onFilter={loadCases} />

      {loading ? (
        <div className="bg-white rounded-lg border border-gray-200 p-8 animate-pulse h-64" />
      ) : (
        <CaseTable cases={caseList} />
      )}
    </div>
  );
}
```
